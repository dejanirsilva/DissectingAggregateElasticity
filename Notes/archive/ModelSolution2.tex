\section{Baseline model}
The aggregate endowment, $Y_t$, follows a geometric Brownian motion:
\begin{equation}\label{eq:agg_endow}
	\frac{d Y_t}{Y_t} = \mu dt + \sigma dZ_t,
\end{equation}
where $\mu\in\mathbb{R}$ and the $1\times d$ vector $\sigma\in\mathbb{R}^d$ are constants.

The economy is populated by $J+1$ types of investors, indexed by $j = 0, 1, \ldots, J$. The mass of type $j$ investors is denoted by $\omega_j$. We assume that investors die with Poisson intensity $\kappa$ and a mass $\kappa \omega_j$ of type $j$ agents are born every period, such that total population is constant and normalized to one. 
Newborn agents inherit the wealth from their parents.
\paragraph{Passive investors.} Investors face different financial constraints. Agents of type $j = 0 $ are \textit{passive investors}, as their portfolio share is given by $\alpha_{0,t} = \overline{\alpha}_{p,t}$, where $\overline{\alpha}_{p,t}$ follows an exogenous square-root process (Cox, Ingersoll, and Ross, 1985):
\begin{equation}\label{eq:alpha_p_process}
	d \overline{\alpha}_{p,t} = \theta_p (\overline{\alpha} - \overline{\alpha}_{p,t}) dt + \sigma_{p} \sqrt{\overline{\alpha}_{p,t}}\, d Z_t,
\end{equation}
where $\overline{\alpha}$ is the long-run mean and $\theta_p$ is the mean-reversion parameter, which controls the speed which passive investors rebalance their portfolios. 

\paragraph{Active investors.} We refer to investors of type $j = 1,\ldots, J$ as \textit{active investors}. Active investors continuously rebalance their portfolios subject to state-dependent leverage constraints. We assume that the maximum portfolio share for an active investor is decreasing in aggregate volatility:
\begin{equation}\label{eq:leverage_constraint}
	\alpha_{j,t} \leq \frac{\overline{\sigma}}{\|\sigma_{R,t}\|},
\end{equation}
where we assume $\overline{\sigma} \geq \|\sigma\|$.

\paragraph{Active investors' problem.} 
Given the homotheticity of preferences, the value function for investor $j$ can be written as
\begin{equation}\label{eq:value-func}
	V_{j,t} = \left( \frac{c_{j,t}}{\rho^\psi} \right)^{\frac{1-\gamma_j}{1-\psi}} \frac{W_{j,t}^{1-\gamma_j}}{1-\gamma_j},
\end{equation}
The HJB equation for agent $ j $ is
\begin{align}\label{eq:HJB}
	0=\max_{C_j,\alpha_j}\,\,&\frac{1}{1-\psi^{-1}}\left[\left(\frac{C_j/W_j}{c_j^{1/(1-\psi)}}\right)^{1-\psi^{-1}}-\rho\right]+
	\left(r+\pi\alpha_{j}-\frac{C_j}{W_j}-\frac{\gamma_j}{2}\alpha_j^2\|\sigma_{R}\|^2\right)\nonumber\\
	&+\frac{1}{1-\psi}\left[\mu_{c_j}+(1-\gamma_j)\alpha_j\sigma_{c_j}\sigma_R'\right]+\frac{1}{2}\left(\frac{1}{1-\psi}\right)\left(\frac{\psi-\gamma_j}{1-\psi}\right)\|\sigma_{c_j}\|^2
\end{align}
%\begin{align}\label{eq:HJB}
%	0=\max_{c_j,\alpha_j}\,\,&\frac{1}{1-\psi^{-1}}(c_j-\rho)+\left(r+\pi\alpha_{j}-c_j-\frac{\gamma_j}{2}\alpha_j^2\|\sigma_{R}\|^2\right)\nonumber\\
%	&+\frac{1}{1-\psi}\left[\frac{c_{j,X}}{c_j}X\mu_X+(1-\gamma_j)\frac{c_{j,X}}{c_j}X\sigma_X\alpha_j\sigma_R+\frac{1}{2}\frac{c_{j,XX}}{c_j}X^2\sigma_X^2\right]\nonumber\\
%	&+\frac{1}{2}\left(\frac{1}{1-\psi}\right)\left[\left(\frac{\psi-\gamma_j}{1-\psi}\right)\left(\frac{c_{j,X}}{c_j}X\sigma_X\right)^2\right]
%\end{align}
where, from the FOC for consumption, $c_{j,t}$ is agent $j$'s consumption-wealth ratio and is a function of the aggregate state variable $X_t$, that is, $c_{j,t} = c_j(X_t)$:
\begin{equation}\label{eq:consumption-wealth}
	c_{j,t}=\frac{C_{j,t}}{W_{j,t}}.
\end{equation}
Also posit the following Ito process for marginal value of wealth
\begin{equation}\label{key}
	\frac{dc_{j,t}}{c_{j,t}}=\mu_{c_j,t}dt+\sigma_{c_j,t}dZ_t.
\end{equation}
We can express the portfolio weight $\alpha_{j,t}$ in terms of $c_{j,t}$ and asset prices: 
\begin{equation}\label{eq:portfolio_share}
	\alpha_{j,t} = \min \left\{ \frac{\pi_t}{\gamma_j \|\sigma_{R,t}\|^2}  + \varsigma_{j,t}\,\,, \,\,\frac{\overline{\sigma}}{\|\sigma_{R,t}\|} \right\}, 
\end{equation}
where $ \pi_t=\mu_{R,t}-r_t $ is the risk premium, and 
\begin{equation}\label{eq:hedging-demand}
	\varsigma_{j,t} \equiv \frac{1-\gamma_j^{-1}}{\psi-1} \frac{\sigma_{c_j,t} \sigma_{R,t}'}{\|\sigma_{R,t}\|^2}
\end{equation} 
is the hedging demand component.

rom the Hamilton-Jacobi-Bellman (HJB) equation, we obtain an expression for $c_{j,t}$:
\begin{equation}\label{eq:xi_j}
	c_{j,t} = \psi \rho + (1-\psi) \left[ r_t + \pi_t \alpha_{j,t} - \frac{\gamma_j}{2} \|\sigma_{R,t}\|^2 \alpha_{j,t}^2\right] + \xi_{j,t},
\end{equation}
where 
\begin{equation}\label{eq:shifter}
	\xi_{j,t} \equiv \mu_{c_j,t} + (1-\gamma_j)  \sigma_{c_j,t} \sigma_{R,t}' \alpha_{j,t} + \frac{\psi - \gamma_j}{1-\psi} \frac{\|\sigma_{c_j,t}\|^2}{2},
\end{equation} corresponds to a forward-looking shifter of the consumption-wealth ratio, which depends on the drift and diffusion of $c_j$.

\paragraph{Pricing condition.} Let $p_t \equiv P_t/Y_t$ denote the price-dividend ratio for the risky asset. Since the expected return is given by $r_t + \pi_t = \frac{1}{p_t} + \mu_{P,t}$, the price-dividend ratio satisfies the condition
\begin{equation}\label{eq:pricing_cond}
	\frac{1}{p_t} = r_t + \pi_t  -(\mu + \mu_{p,t} + \sigma \sigma_{p,t}'),
\end{equation}
where $\sigma_{R,t} = \sigma + \sigma_{p,t}$, $(\mu_{p,t}, \sigma_{p,t})$ is given by Ito's lemma, and we used $\mu_{P,t} =\mu + \mu_{p,t} + \sigma \sigma_{p,t}' $.

\paragraph{Aggregate state variable.} Let $x_j$ be the wealth share of type-$j$ investors:
\begin{equation}
	x_{j,t} \equiv \frac{\omega_j W_{j,t}}{P_t}.\nonumber
\end{equation}
We define the aggregate state variable as 
\begin{equation}\label{eq:agg-state-var}
	X_t = (x_t, \overline{\alpha}_{p,t}), 
\end{equation}
where $x_t \equiv \left(x_{1,t}, x_{2,t}, \ldots, x_{J,t}\right )$. 

The law of motion for the exogenous state variable $\overline{\alpha}_{p,t}$ is given in Equation~\eqref{eq:alpha_p_process}.


From Ito's lemma, the law of motion of $x_{j,t}$ is given by
\begin{small}
	\begin{align*}
		\frac{d x_{j,t}}{x_{j,t}} &= \left( r_t + \pi_t \alpha_{j,t} - c_{j,t} -\mu - \mu_{p,t} - \sigma \sigma_{p,t}'  +(1- \alpha_{j,t}) \|\sigma_{R,t}\|^2  +\kappa \frac{\omega_j-x_{j,t}}{x_{j,t}}  \right) dt + \left (\alpha_{j,t} - 1\right ) \sigma_{R,t} d Z_t.
	\end{align*}
\end{small}


\paragraph{Market clearing conditions.} 
Let $\mathcal{J}_t^u \subseteq \{1,2,\ldots, J\}$ and $\mathcal{J}_t^c \subseteq \{1,2,\ldots, J\}$ denote the set of unconstrained and constrained active investors at time $t$, respectively.
Define the wealth share of unconstrained investors as $x_{u,t} \equiv \sum_{j\in \mathcal{J}_t^u} x_{j,t}$, and the wealth share of constrained investors as  $x_{c,t} \equiv \sum_{j\in \mathcal{J}_t^c} x_{j,t}$
The market clearing condition for the risky asset is:
\begin{equation}\label{eq:risky-asset-mc2}
	\sum_{j=0}^Jx_{j,t}\alpha_{j,t}=1.
\end{equation}
From the market clearing for the risky asset above, we obtain
\begin{equation}\label{eq:risky-asset-mc}
	\underbrace{\sum_{j \in \mathcal{J}_t^u} x_{j,t} \alpha_{j,t} }_{\substack{\textbf{active unconstrained}\\
			\textbf{demand}}} = \underbrace{1 - x_{0,t} \overline{\alpha}_{p,t} - x_{c,t} \overline{\alpha}_{c,t} }_{\textbf{net supply}},
\end{equation}
where $\overline{\alpha}_{c,t} \equiv\frac{\overline{\sigma}}{\|\sigma_{R,t}\|}$ is the portfolio share of constrained agents. 

We can write the market-clearing condition for goods as follows:
\begin{equation}\label{eq:goods-mc}
	\sum_{j=0}^J x_j c_{j,t} = \frac{1}{p_t}.
\end{equation}

The risk-free bond market clears by Walras's law:
\begin{equation}\label{eq:risk-free-mc}
	\sum_{j=0}^Jx_{j,t}(1-\alpha_{j,t})=0.
\end{equation}


\paragraph{Risk premium and interest rate.} 


Combining the market clearing condition in Equation~\eqref{eq:risky-asset-mc} with the optimal portfolio of active investors in \eqref{eq:portfolio_share}, we obtain an expression for the risk premium:
\begin{equation}\label{eq:sharpe_ratio}
	\pi_{t} =  \frac{\gamma_{u,t} \|\sigma_{R,t}\|^2}{x_{u,t}} \left[ 1 - x_{0,t} \overline{\alpha}_{p,t} - x_{c,t} \overline{\alpha}_{c,t} - x_{u,t} \varsigma_{t} \right] ,
\end{equation}
where $\gamma_{u,t} \equiv \left[ \sum_{j\in\mathcal{J}_t^u} \frac{x_{j,t}}{x_{u,t}} \frac{1}{\gamma_j}\right]^{-1}$ denotes the average risk aversion, $ \overline{\alpha}_{c,t}=\frac{\overline{\sigma}}{\|\sigma_{R,t}\|} $ is the portfolio share of constrained agents, and $\varsigma_t \equiv \sum_{j\in \mathcal{J}_t^u} \frac{x_{j,t}}{x_{u,t}} \varsigma_{j,t} $ the average hedging demand of unconstrained investors.

%Equation~\eqref{eq:sharpe_ratio} captures the effect of heterogeneity, passive investment, and leverage constraints on the risk premium. Keeping everything else constant, an increase in the average risk aversion of unconstrained agents (higher $\gamma_{u,t}$), a reduction in passive investors' demand (lower $\overline{\alpha}_{p,t}$), a reduction in the risk bearing capacity of constrained investor (lower $\overline{\alpha}_{c,t}$), or a reduction in hedging demands (lower $\varsigma_t$) all tend to increase the risk premium. 

From the Equations~\eqref{eq:goods-mc} and \eqref{eq:xi_j}, we obtain
\begin{align}\label{eq:interest_rate}
	r_t  &  =  \rho  +\psi^{-1} \mu_{P,t} + \left (1-\psi^{-1}\right )  \sum_{j=0}^J x_{j,t}\frac{\gamma_j \alpha_{j,t}^2}{2} ||\sigma_{R,t}||^2- \pi_t +\psi^{-1} \xi_t,
\end{align}
where $\xi_t \equiv \sum_{j=0}^J x_{j,t} \xi_{j,t}$.

\paragraph{Dynamics of consumption-wealth and price-dividend ratios.}
By Ito's lemma we can get drift and diffusion of consumption-wealth ratio for agent $ j $:
\begin{align}
	\mu_{c_{j},t}&=\frac{c_{j,x}(X_t)}{c_j(X_t)}\mu_{x,t}+\frac{c_{j,\overline{\alpha}_p}(X_t)}{c_j(X_t)}\theta_p\left(\overline{\alpha}-\overline{\alpha}_{p,t}\right)+\frac{1}{2}\frac{c_{j,xx}(X_t)}{c_j(X_t)}\|\sigma_{x,t}\|^2,\\
	\sigma_{c_j,t}&=\frac{c_{j,x}(X_t)}{c_j(X_t)} \sigma_{x,t} + \frac{c_{j,{\overline{\alpha}_p}}(X_t)}{c_j(X_t)} \sigma_{p} \sqrt{\overline{\alpha}_{p,t}},
\end{align}
By Ito's lemma we can get drift and diffusion of the price-dividend ratio $ p $:
\begin{align}
	\mu_{p,t}&=\frac{p_x(X_t)}{p(X_t)}\mu_{x,t}+\frac{p_{\overline{\alpha}_p}(X_t)}{p(X_t)}\theta_p\left(\overline{\alpha}-\overline{\alpha}_{p,t}\right)+\frac{1}{2}\frac{p_{xx}(X_t)}{p(X_t)}\|\sigma_{x,t}\|^2,\\
	\sigma_{p,t} &= \frac{p_x(X_t)}{p(X_t)} \sigma_{x,t} + \frac{p_{\overline{\alpha}_p}(X_t)}{p(X_t)} \sigma_{p} \sqrt{\overline{\alpha}_{p,t}}. 
\end{align}



\section{Solution Method}

\paragraph{Endogenous volatility and portfolios.} The diffusion term for the $J\times1$ vector $x$ is given by
\begin{equation}
    \sigma_{x,t} = \Tilde{\sigma}_{x,t} \sigma_{R,t},
\end{equation}
where $\sigma_{x,t}$ is a $J\times 2$ matrix, $\Tilde{\sigma}_{x,t}$ is a $J\times 1$ vector, and $\sigma_{R,t}$ is a $1\times 2$ (row) vector. The $j$-th element of $\Tilde{\sigma}_{x,t}$ is given by 
\begin{equation}
    \widetilde{\sigma}_{x,j}=x_{j,t}(\alpha_{j,t}-1).
\end{equation}

Using the expression for the risk premium, we can the portfolio share of investor $j =1,\ldots, J$ as follows:
\begin{equation}
    \alpha_{j,t} = \frac{\gamma_{u,t}}{\gamma_j} \frac{1-x_{0,t} \overline{\alpha}_{p,t}}{x_{u,t}} - \frac{\gamma_{u,t}}{\gamma_j} \sum_{j=1} \frac{x_{i,t}}{x_{u,t}} \varsigma_{i,t} + \varsigma_{j,t},
\end{equation}
where $x_{u,t} = \sum_{j=1}^J x_{j,t} $ and $x_{0,t} = 1-x_{u,t}$.

The hedging demand is given 
\begin{equation}
    \varsigma_{j,t} = \frac{1-\gamma_j^{-1}}{\psi-1} \frac{\sigma_{c_j,t} \sigma_{R,t}'}{\sigma_{R,t}\sigma_{R,t}'},
\end{equation}
where $\sigma_{c_j,t} = \frac{c_{j,x}}{c_{j,t}} \sigma_{x,t} + \frac{c_{j,\overline{\alpha}_p}}{c_{j,t}} \sigma_{\overline{\alpha}_p,t}$.

Using the expression for $\sigma_{x,t}$, we can write the hedging demand as follows
\begin{equation}
    \varsigma_{j,t} = \frac{1-\gamma_j^{-1}}{\psi-1} \left[\frac{c_{j,x}}{c_{j,t}} \tilde{\sigma}_{x,t}+\frac{c_{j,\overline{\alpha}_p}}{c_{j,t}} \frac{\sigma_{\overline{\alpha}_p,t} \sigma_{R,t}'}{\sigma_{R,t}\sigma_{R,t}'}\right].
\end{equation}

Combining the previous expressions, we can write $\widetilde{\sigma}_{x,t}$ as follows:
\begin{align}
    \widetilde{\sigma}_{x,j,t} &= x_{j,t} \left[ \frac{\gamma_{u,t}}{\gamma_j} \frac{1-x_{0,t} \overline{\alpha}_{p,t}}{x_{u,t}} - \frac{\gamma_{u,t}}{\gamma_j} \sum_{i=1} \frac{x_{i,t}}{x_{u,t}} \frac{1-\gamma_i^{-1}}{\psi-1} \left[\frac{c_{i,x}}{c_{i,t}} \tilde{\sigma}_{x,t}+\frac{c_{i,\overline{\alpha}_p}}{c_{i,t}} \frac{\sigma_{\overline{\alpha}_p,t} \sigma_{R,t}'}{\sigma_{R,t}\sigma_{R,t}'}\right] \right.\nonumber\\
    &\left.+ \frac{1-\gamma_j^{-1}}{\psi-1} \left[\frac{c_{j,x}}{c_{j,t}} \tilde{\sigma}_{x,t}+\frac{c_{j,\overline{\alpha}_p}}{c_{j,t}} \frac{\sigma_{\overline{\alpha}_p,t} \sigma_{R,t}'}{\sigma_{R,t}\sigma_{R,t}'}\right]-1\right].
\end{align}

In matrix form, we expression can be written as follows:
\begin{equation}
    \widetilde{\sigma}_{x} = a_{0} + a_1 \frac{\sigma_{\overline{\alpha}_p} \sigma_{R}'}{\sigma_{R}\sigma_{R}'} + A \widetilde{\sigma}_x,
\end{equation}
where $a_{0,j} = x_{j,t} \left( \frac{\gamma_{u,t}}{\gamma_j} \frac{1-x_{0,t} \overline{\alpha}_{p,t}}{x_{u,t}}  - 1\right) $, $a_{1,j} = x_{j,t} \left[ \frac{1-\gamma_j^{-1}}{\psi-1} \frac{c_{j,\overline{\alpha}_p}}{c_{j,t}} -  
\frac{\gamma_{u,t}}{\gamma_j} \sum_{j=1} \frac{x_{i,t}}{x_{u,t}} \frac{1-\gamma_i^{-1}}{\psi-1} \frac{c_{i,\overline{\alpha}_p}}{c_{i,t}}
\right]$, and the $j$-th row of matrix $A$ is given by $A_j = x_{j,t} \left[ \frac{1-\gamma_j^{-1}}{\psi-1} \frac{c_{j,x}}{c_{j,t}} - \frac{\gamma_{u,t}}{\gamma_j} \sum_{i=1} \frac{x_{i,t}}{x_{u,t}} \frac{1-\gamma_i^{-1}}{\psi-1} \frac{c_{i,x}}{c_{i,t}} \right] $.

Therefore, we can write $\widetilde{\sigma}_x$ as follows:
\begin{equation}
    \widetilde{\sigma}_{x} = [I-A]^{-1} \left[ a_{0} + a_1 \frac{\sigma_{\overline{\alpha}_p} \sigma_{R}'}{\sigma_{R}\sigma_{R}'} \right],
\end{equation}
assuming that the matrix $I-A$ is invertible.

The diffusion term for returns is given by
\begin{equation}
\sigma_R = \sigma + \frac{p_x}{p} \sigma_x + \frac{p_{\overline{\alpha}_p}}{p} \sigma_{\overline{\alpha}_p}.
\end{equation}

Using the expression for $\sigma_{x}$, we can rewrite the expression above as follows:
\begin{equation}
    \sigma_{R} = \frac{\sigma + \frac{p_{\overline{\alpha}_p}}{p} \sigma_{\overline{\alpha}_p}}{1-\frac{p_x}{p} \widetilde{\sigma}_x}. 
\end{equation}

Notice that this implies that the following term is given by
\begin{equation}
    \frac{\sigma_{\overline{\alpha}_p} \sigma_{R}'}{\sigma_{R}\sigma_{R}'} = 
    \frac{\sigma_{\overline{\alpha}_p}\left(\sigma' + \frac{p_{\overline{\alpha}_p}}{p} \sigma_{\overline{\alpha}_p}'\right)}{\left(\sigma + \frac{p_{\overline{\alpha}_p}}{p} \sigma_{\overline{\alpha}_p}\right)\left(\sigma' + \frac{p_{\overline{\alpha}_p}}{p} \sigma_{\overline{\alpha}_p}'\right)} \left[ 1-\frac{p_x}{p} \widetilde{\sigma}_x \right].
\end{equation}

We can then solve for $\frac{p_x}{p} \widetilde{\sigma}_x$:
\begin{equation}
    \frac{p_x}{p} \widetilde{\sigma}_x = \frac{\frac{p_x}{p} [I-A]^{-1} \left[a_0+a_1 \frac{\sigma_{\overline{\alpha}_p}\left(\sigma' + \frac{p_{\overline{\alpha}_p}}{p} \sigma_{\overline{\alpha}_p}'\right)}{\left(\sigma + \frac{p_{\overline{\alpha}_p}}{p} \sigma_{\overline{\alpha}_p}\right)\left(\sigma' + \frac{p_{\overline{\alpha}_p}}{p} \sigma_{\overline{\alpha}_p}'\right)}\right]}{1+\frac{p_x}{p} [I-A]^{-1} a_1 \frac{\sigma_{\overline{\alpha}_p}\left(\sigma' + \frac{p_{\overline{\alpha}_p}}{p} \sigma_{\overline{\alpha}_p}'\right)}{\left(\sigma + \frac{p_{\overline{\alpha}_p}}{p} \sigma_{\overline{\alpha}_p}\right)\left(\sigma' + \frac{p_{\overline{\alpha}_p}}{p} \sigma_{\overline{\alpha}_p}'\right)} }.
\end{equation}

The expression above gives us the scalar $\frac{p_x}{p} \widetilde{\sigma}_x$. Given this scalar, we can solve for $\sigma_R$ and $\widetilde{\sigma}_x$, so we can obtain $\sigma_x$. This allow us to solve for the hedging demands $\varsigma_{j,t}$ and portfolio choices $\alpha_{j,t}$. Given the hedging demands and the volatility, we can solve for $\pi_t$.











To solve for the equilibrium of this economy, we need to find consumption-wealth ratio functions $ c_j(X) $ and price-dividend ratio $ p(X) $. Given these two functions, we can solve for all remaining equilibrium objects as follows:
